{
    ios::sync_with_stdio(false);

    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        fun();
    }
    return 0;
}
void fun()
{
    int n;
    cin>>n;
    long int arr[n];
    double sum=0,mean;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
        sum+=arr[i];
    }
    mean=sum/n;
    //sort(arr,arr+n);
    long int steal;
    int ctr=0;
    for(int i=0;i<n;i++)
    {
        if(((sum-double(arr[i]))/(n-1))==mean)
        {steal=i+1;
         break;}
        else
        {
            ctr++;
        }
    }
    if(ctr==n)
    cout<<"Impossible\n";
    else
    {
        cout<<steal<<"\n";
    }
}
